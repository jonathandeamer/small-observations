{{/* Renders a responsive for a post's photo. Reads `.Params.photo` (path relative to assets/img/) and `.Params.alt`. Above-the-fold callers pass `eager` in the params dict to skip lazy-loading. "eager" false) }} */}} {{ $page := .page }} {{ $eager := .eager | default false }} {{ $rel := printf "img/%s" $page.Params.photo }} {{ $orig := resources.Get $rel }} {{ if not $orig }} {{ errorf "photo not found in assets/: %s (post: %s)" $rel $page.RelPermalink }} {{ end }} {{ $jpeg600 := $orig.Resize "600x q82" }} {{ $jpeg1000 := $orig.Resize "1000x q82" }} {{ $jpeg1600 := $orig.Resize "1600x q82" }} {{ $webp600 := $orig.Process "resize 600x webp q80" }} {{ $webp1000 := $orig.Process "resize 1000x webp q80" }} {{ $webp1600 := $orig.Process "resize 1600x webp q80" }}
